% Nouvelle commande pour lire un fichier texte et échapper les caractères spéciaux
\newcommand{\readtextfile}[1]{%
    \begingroup
    \catcode`\#=12 % Make # a non-special character
    \catcode`\$=12 % Make $ a non-special character
    \catcode`\_=12 % Make _ a non-special character
    \catcode`\&=12 % Make & a non-special character
    \catcode`\~=12 % Make ~ a non-special character
    \catcode`\%=12 % Make % a non-special character
    \catcode`\^=12 % Make ^ a non-special character
    \input{#1}%
    \endgroup
}
